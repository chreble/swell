 10 Swell.Core.Class({
 11     
 12     name      : 'Ajax',
 13     namespace : 'Lib.IO',
 14     inherits  : Swell.Core.CustomEventModel,
 15     functions : function() {
 16         /**
 17          * Do not expose the internal functions below
 18          * to the public API
 19          *
 20          * @private
 21         */
 22         var _getXhrObject = function() {
 23             var _xmlHttp = null;
 24             // Testing if XMLHttpRequest is native (that should be the case for every modern browser)
 25             if(window.XMLHttpRequest) {
 26                 _xmlHttp = new XMLHttpRequest();
 27             } else {
 28                 // Wrap variable assignement into a try/catch block so it will prevent
 29                 // javascript errors if the user has disabled ActiveX Controls
 30                 try {
 31                     // This is IE Browsers < 7 > 5.0
 32                     _xmlHttp = new ActiveXObject('MSXML2.XMLHTTP.3.0'); // Use the latest and greatest one :D
 33                 } catch (e) {
 34                     _xmlHttp = null;
 35                 }
 36             }
 37             return _xmlHttp;
 38         };
 39         
 40         /** @lends Swell.Lib.IO.Ajax.prototype */ 
 41         return {
 42             /**
 43              * @property LOADING
 44              * @description used internally for checking if readyState is equal to 1 (loading)
 45              * @static
 46             */
 47             LOADING       : 1,
 48             /**
 49              * @property COMPLETED
 50              * @description used internally for checking if readyState is equal to 4 (completed)
 51             */
 52             COMPLETED     : 4,
 53             /**
 54              * @property {XMLHttpRequest|NULL} Connection object
 55              * @description Internal pointer on native XHR Object
 56             */
 57             xhr    : null,
 58             /** 
 59              * @constructs
 60              * @augments Swell.Core.CustomEventModel
 61             */
 62             construct : function() {
 63                 var _xhr  = _getXhrObject();
 64                 // Exit nicely if XHR object detection failed
 65                 if(!_xhr) {
 66                     return false;
 67                 }
 68                 // Assign a pointer to the current XHR object, this will help us later in the script
 69                 this.xhr = _xhr;
 70                 this.createEvents();
 71             },
 72             
 73             /**
 74              * Initiates an XML HTTP Request
 75              * Note : This implementation provides only asynchronous XMLHttpRequests because, 
 76                       due to the inherently asynchronous nature of networking, 
 77                       there are various ways memory and events can leak when using synchronous requests.
 78              *
 79              * @param {String} m http transaction method either GET or POST
 80              * @param {String} url full path to the remote page or resource to retrieve through XHR
 81              * @param {Function} fn callback function
 82              * @see http://www.quirksmode.org/blog/archives/2005/09/xmlhttp_notes_r_2.html
 83             */
 84             request : function(m, url, fn, scope, args) {
 85                 
 86                 var _args       = args || null,
 87                     _scope      = scope || null,
 88                     that    = this;
 89 
 90                 // Exit nicely if fn is not a function
 91                 if(!Swell.Core.isFunction(fn)) {
 92                     return false;
 93                 }
 94                 // Exit nicely if m is not get or post (case insensitive)
 95                 if(!/^get$|^post$/i.test(m)) {
 96                     return false;
 97                 }
 98                 
 99                 this.xhr.onreadystatechange = function(e) {
100                     if(window.event) {
101                         e = window.event;
102                     }
103                     if(that.xhr.readyState === that.LOADING) {
104                         that.fireEvent('onProgress');
105                     }
106                     else if(that.xhr.readyState === that.COMPLETED) {
107                         // Detach all subscribers of progress event
108                         that.unsubscribe('onProgress');
109                         // Fire complete event
110                         that.fireEvent('onComplete');
111                         // Execute callback function with the given scope and args
112                         fn.call(_scope, that, _args);
113                     }
114                 };
115                 
116                 // Fire onInitiate event
117                 this.fireEvent('onInitiate');
118                 
119                 // Open connection
120                 this.xhr.open(m.toUpperCase(), url, true);
121                 this.xhr.send(null);
122             },
123             
124             /**
125              * Initialize events of the class
126             */
127             createEvents : function() {
128                 this.onInitiate();
129                 this.onProgress();
130                 this.onComplete();
131             },
132             
133             /**
134              * @event onInitiate
135              * @description Fires when XmlHttpRequest is initiated
136             */
137             onInitiate : function() {
138                 this.createEvent('onInitiate');
139             },
140             /**
141              * @event onProgress
142              * @description Fires when XmlHttpRequest is currently processed
143             */
144             onProgress : function() {
145                 this.createEvent('onProgress');
146             },
147             /**
148              * @event onComplete
149              * @description Fires when XmlHttpRequest is completed
150             */
151             onComplete : function() {
152                 this.createEvent('onComplete');
153             }
154         };
155     }()
156 });
157 
158 // Set some shorthands for the class
159 Swell.alias(Swell.Lib.IO.Ajax, 'Ajax', 'Connection');
